# Installation Guide for Lawglance Dependencies

# 1. Install core Python packages
pip install --user langchain transformers nltk spacy textstat
pip install --user faiss-cpu python-docx networkx scikit-learn numpy

# 2. Install SpaCy language models
python -m spacy download en_core_web_sm

# 3. Download required NLTK resources
python -c "import nltk; nltk.download('punkt'); nltk.download('stopwords'); nltk.download('wordnet')"

# 4. Verify installation
python -c "import langchain; import transformers; import spacy; import textstat; import docx; import networkx; import sklearn; import numpy; import nltk; print('All dependencies successfully imported!')"
